What is a Multiset?

A multiset (or bag) is a generalization of a set that allows multiple instances of its elements. Unlike a regular set where each element must be unique, a multiset tracks how many times each element appears.

Multiset Implementation

Here's a simple JavaScript implementation of a Multiset:


class Multiset {
    constructor() {
        this.elements = new Map(); // Stores elements and their counts
    }
    
    // Add an element to the multiset
    add(element) {
        if (this.elements.has(element)) {
            this.elements.set(element, this.elements.get(element) + 1);
        } else {
            this.elements.set(element, 1);
        }
    }
    
    // Remove an element from the multiset
    remove(element) {
        if (!this.elements.has(element)) return false;
        
        const count = this.elements.get(element);
        if (count === 1) {
            this.elements.delete(element);
        } else {
            this.elements.set(element, count - 1);
        }
        return true;
    }
    
    // Get the count of an element
    count(element) {
        return this.elements.get(element) || 0;
    }
    
    // Get all unique elements
    uniqueElements() {
        return Array.from(this.elements.keys());
    }
    
    // Get total size including duplicates
    size() {
        let total = 0;
        for (const count of this.elements.values()) {
            total += count;
        }
        return total;
    }
    
    // Check if multiset is empty
    isEmpty() {
        return this.elements.size === 0;
    }
    
    // Convert to string representation
    toString() {
        let result = [];
        for (const [element, count] of this.elements) {
            result.push(`${element} (${count})`);
        }
        return `Multiset {${result.join(', ')}}`;
    }
}
