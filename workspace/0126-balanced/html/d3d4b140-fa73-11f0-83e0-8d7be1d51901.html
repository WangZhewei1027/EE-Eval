Stack API (JavaScript)

Implementation built into this demo. See the code and the array view below.
Capacity: ∞

Operations:

  • push(item) — add item to the top
  • pop() — remove and return the top item
  • peek() — view the top item without removing
  • isEmpty() / size() — check stack state
  • clear() — empty the stack
// Simple Stack class used in this demo
class Stack {
  constructor(max = Infinity){
    this._items = [];
    this.max = max;
  }
  push(item){
    if(this._items.length >= this.max) throw new Error("Stack overflow");
    this._items.push(item);
  }
  pop(){
    return this._items.pop();
  }
  peek(){
    return this._items.length === 0 ? undefined : this._items[this._items.length-1];
  }
  size(){ return this._items.length; }
  isEmpty(){ return this._items.length === 0; }
  clear(){ this._items.length = 0; }
}
