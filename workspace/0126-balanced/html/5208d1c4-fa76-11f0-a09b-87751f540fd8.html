Counting Sort

Counting Sort is a sorting algorithm that sorts data with integer keys by grouping keys by range and then sorting within each group.

        
        function countingSort(arr) {
            let max = Math.max(...arr);
            let min = Math.min(...arr);
            let n = arr.length;
            let output = new Array(n).fill(0);
            let count = new Array(max + 1).fill(0);

            for (let i = 0; i < n; i++) {
                count[arr[i]]++;
            }

            for (let i = 1; i <= max; i++) {
                count[i] += count[i - 1];
            }

            for (let i = n - 1; i >= 0; i--) {
                output[count[arr[i]] - 1] = arr[i];
                count[arr[i]]--;
            }

            for (let i = 0; i < n; i++) {
                arr[i] = output[i];
            }
            return arr;
        }
