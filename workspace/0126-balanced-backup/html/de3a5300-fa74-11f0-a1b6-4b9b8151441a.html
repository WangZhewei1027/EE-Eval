JavaScript Arrays Demonstration

1. Creating Arrays

// Array literal syntax
let fruits = ['Apple', 'Banana', 'Orange'];

// Array constructor
let numbers = new Array(1, 2, 3, 4, 5);

// Empty array
let emptyArray = [];

// Array with mixed types
let mixedArray = [1, 'two', true, {name: 'John'}, [5, 6]];

2. Accessing Array Elements

let colors = ['Red', 'Green', 'Blue'];

// Access by index (0-based)
let firstColor = colors[0];  // 'Red'
let secondColor = colors[1]; // 'Green'

// Access last element
let lastColor = colors[colors.length - 1]; // 'Blue'

3. Modifying Arrays

let animals = ['Dog', 'Cat', 'Elephant'];

// Modify an element
animals[1] = 'Lion'; // ['Dog', 'Lion', 'Elephant']

// Add to end
animals.push('Giraffe'); // ['Dog', 'Lion', 'Elephant', 'Giraffe']

// Remove from end
let lastAnimal = animals.pop(); // 'Giraffe', array is now ['Dog', 'Lion', 'Elephant']

// Add to beginning
animals.unshift('Zebra'); // ['Zebra', 'Dog', 'Lion', 'Elephant']

// Remove from beginning
let firstAnimal = animals.shift(); // 'Zebra', array is now ['Dog', 'Lion', 'Elephant']

4. Array Methods

let numbers = [1, 2, 3, 4, 5];

// Join elements into a string
let joined = numbers.join('-'); // "1-2-3-4-5"

// Slice - get a portion of the array
let sliced = numbers.slice(1, 4); // [2, 3, 4]

// Splice - add/remove elements
let removed = numbers.splice(2, 2, 6, 7); // removes [3, 4], adds 6, 7
// numbers is now [1, 2, 6, 7, 5]

// Concatenate arrays
let moreNumbers = [8, 9];
let combined = numbers.concat(moreNumbers); // [1, 2, 6, 7, 5, 8, 9]

// Find index of element
let index = numbers.indexOf(6); // 2

5. Iterating Over Arrays

let languages = ['JavaScript', 'Python', 'Java', 'C#'];

// for loop
for (let i = 0; i < languages.length; i++) {
    console.log(languages[i]);
}

// for...of loop
for (let lang of languages) {
    console.log(lang);
}

// forEach method
languages.forEach(function(lang, index) {
    console.log(`${index}: ${lang}`);
});

// map method
let lengths = languages.map(lang => lang.length);
// lengths is [10, 6, 4, 2]

6. Array Destructuring

let rgb = ['Red', 'Green', 'Blue'];

// Basic destructuring
let [first, second, third] = rgb;
// first = 'Red', second = 'Green', third = 'Blue'

// Skipping elements
let [r, , b] = rgb;
// r = 'Red', b = 'Blue'

// Default values
let [color1, color2, color3, color4 = 'Black'] = rgb;
// color4 defaults to 'Black' if not present
