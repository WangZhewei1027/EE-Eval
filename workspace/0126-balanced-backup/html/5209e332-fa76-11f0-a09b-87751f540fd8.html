Time Complexity

Time complexity is a measure of how long an algorithm takes to complete as a function of the size of the input.

Big O Notation

Big O notation is a way to describe the time complexity of an algorithm. It is usually expressed as O(n), O(log n), O(n log n), etc.

Let's create a simple JavaScript function to demonstrate time complexity.

Example 1: Bubble Sort

        
        function bubbleSort(arr) {
            let n = arr.length;
            for (let i = 0; i < n; i++) {
                for (let j = 0; j < n - i - 1; j++) {
                    if (arr[j] > arr[j + 1]) {
                        [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
                    }
                }
            }
            return arr;
        }

Example 2: Binary Search

        
        function binarySearch(arr, target) {
            let left = 0;
            let right = arr.length - 1;
            while (left <= right) {
                let mid = Math.floor((left + right) / 2);
                if (arr[mid] === target) {
                    return mid;
                } else if (arr[mid] < target) {
                    left = mid + 1;
                } else {
                    right = mid - 1;
                }
            }
            return -1;
        }

Example 3: Fibonacci Sequence

        
        function fibonacci(n) {
            if (n <= 1) {
                return n;
            } else {
                return fibonacci(n - 1) + fibonacci(n - 2);
            }
        }

Example 4: Matrix Multiplication

        
        function matrixMultiply(a, b) {
            let m = a.length;
            let n = b.length;
            let result = new Array(m).fill(0).map(() => new Array(n).fill(0));
            for (let i = 0; i < m; i++) {
                for (let j = 0; j < n; j++) {
                    for (let k = 0; k < n; k++) {
                        result[i][j] += a[i][k] * b[k][j];
                    }
                }
            }
            return result;
        }

Example 5: Counting Sort

        
        function countingSort(arr) {
            let max = Math.max(...arr);
            let count = new Array(max + 1).fill(0);
            for (let i = 0; i < arr.length; i++) {
                count[arr[i]]++;
            }
            for (let i = 1; i < count.length; i++) {
                count[i] += count[i - 1];
            }
            let output = new Array(arr.length).fill(0);
            for (let i = arr.length - 1; i >= 0; i--) {
                output[count[arr[i]] - 1] = arr[i];
                count[arr[i]]--;
            }
            return output;
        }

Example 6: Merge Sort

        
        function mergeSort(arr) {
            if (arr.length <= 1) {
                return arr;
            }
            let mid = Math.floor(arr.length / 2);
            let left = mergeSort(arr.slice(0, mid));
            let right = mergeSort(arr.slice(mid));
            return merge(left, right);
        }
        function merge(left, right) {
            let result = new Array(left.length + right.length);
            let i = 0, j = 0, k = 0;
            while (i < left.length && j < right.length) {
                if (left[i] <= right[j]) {
                    result[k++] = left[i++];
                } else {
                    result[k++] = right[j++];
                }
            }
            while (i < left.length) {
                result[k++] = left[i++];
            }
            while (j < right.length) {
                result[k++] = right[j++];
            }
            return result;
        }

Example 7: Quick Sort

        
        function quickSort(arr) {
            if (arr.length <= 1) {
                return arr;
            }
            let pivot = arr[Math.floor(arr.length / 2)];
            let left = arr.filter(x => x < pivot);
            let middle = arr.filter(x => x === pivot);
            let right = arr.filter(x => x > pivot);
            return quickSort(left).concat(middle).concat(quickSort(right));
        }

Example 8: Radix Sort

        
        function radixSort(arr) {
            if (arr.length <= 1) {
                return arr;
            }
            let max = Math.max(...arr);
            let digits = 1;
            while (max / digits > 0) {
                let bucket = new Array(max + 1).fill(0);
                for (let i = 0; i < arr.length; i++) {
                    let index = (arr[i] / digits) % 10;
                    bucket[index]++;
                }
                for (let i = 1; i < bucket.length; i++) {
                    bucket[i] += bucket[i - 1];
                }
                let output = new Array(arr.length);
                for (let i = 0; i < arr.length; i++) {
                    output[bucket[arr[i] / digits] - 1] = arr[i];
                    bucket[arr[i] / digits]--;
                }
                arr = output;
                digits *= 10;
            }
            return arr;
        }

Example 13: Time Complexity Analysis

Time complexity analysis is a crucial step in understanding the performance of algorithms. Here's a brief analysis of the time complexity of each example:

  • Bubble Sort: O(n^2)
  • Binary Search: O(log n)
  • Fibonacci Sequence: O(n)
  • Matrix Multiplication: O(n^3)
  • Counting Sort: O(n + k)
  • Merge Sort: O(n log n)
  • Quick Sort: O(n log n) on average, O(n^2) in the worst case
  • Radix Sort: O(nk)
  • Bubble Sort with Optimizations: O(n^2)
  • Merge Sort with Optimizations: O(n log n)
  • Quick Sort with Optimizations: O(n log n) on average, O(n^2) in the worst case
  • Radix Sort with Optimizations: O(nk)
