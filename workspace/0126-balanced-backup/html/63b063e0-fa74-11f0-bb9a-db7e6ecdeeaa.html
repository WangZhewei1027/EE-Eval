Multiset (Bag) Demonstration

A Multiset is like a set but allows multiple occurrences of the same element. This example implements a simple multiset class in JavaScript and lets you add, remove, and query elements with multiplicities.

Multiset Class Implementation


// Multiset class stores elements with counts
class Multiset {
  constructor() {
    this.map = new Map(); // element -> count
  }

  add(element, count = 1) {
    if(count <= 0) return;
    this.map.set(element, (this.map.get(element) || 0) + count);
  }

  remove(element, count = 1) {
    if(!this.map.has(element)) return;
    const currentCount = this.map.get(element);
    if(count >= currentCount) {
      this.map.delete(element);
    } else {
      this.map.set(element, currentCount - count);
    }
  }

  count(element) {
    return this.map.get(element) || 0;
  }

  has(element) {
    return this.map.has(element);
  }

  // Total number of elements including duplicates
  size() {
    let total = 0;
    for(const count of this.map.values()) {
      total += count;
    }
    return total;
  }

  // Number of distinct elements
  distinctSize() {
    return this.map.size;
  }

  // Returns an array of [element, count]
  entries() {
    return Array.from(this.map.entries());
  }

  clear() {
    this.map.clear();
  }

  // Convert to string format like: a:3, b:1, c:5
  toString() {
    if(this.map.size === 0) return "(empty)";
    return Array.from(this.map.entries())
      .map(([el, count]) => `${el}: ${count}`)
      .join(", ");
  }
}
