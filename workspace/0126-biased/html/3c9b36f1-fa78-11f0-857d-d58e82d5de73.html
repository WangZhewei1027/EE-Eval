// Sample interpreter processing a simple expression
function interpret() {
  const source = '2 + 3 * (7 - 4)';
  const tokens = lexer(source);
  const ast = parser(tokens);
  return evaluate(ast);
}
The Interpreter reads source code, transforms it into tokens, parses to an Abstract Syntax Tree (AST), then evaluates expressions to produce results.
