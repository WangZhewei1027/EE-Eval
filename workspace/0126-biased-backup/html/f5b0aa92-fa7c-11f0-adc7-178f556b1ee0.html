Heap Sort

Heap Sort is a comparison-based sorting algorithm that uses a binary heap data structure.

It is a popular sorting algorithm due to its simplicity, efficiency, and stability.

Here is a step-by-step textual description of the Heap Sort algorithm:

  1. Start with a large, unsorted array of elements.
  2. Build a max heap from the unsorted array.
  3. Remove the largest element (the root) from the heap and place it at the end of the sorted array.
  4. Replace the root with the second-largest element from the heap and heapify the reduced heap.
  5. Repeat steps 4 and 5 until the heap is empty.

Here is an example of how Heap Sort works on an array of integers:

            
                let arr = [64, 34, 25, 12, 22, 11, 90];
                console.log("Before Heap Sort:");
                console.log(arr);
                heapSort(arr);
                console.log("After Heap Sort:");
                console.log(arr);

Here is a JavaScript implementation of the Heap Sort algorithm:

            function heapify(arr, n, i) {
                let largest = i;
                let left = 2 * i + 1;
                let right = 2 * i + 2;
                if (left < n && arr[left] > arr[largest]) {
                    largest = left;
                }
                if (right < n && arr[right] > arr[largest]) {
                    largest = right;
                }
                if (largest !== i) {
                    [arr[i], arr[largest]] = [arr[largest], arr[i]];
                    heapify(arr, n, largest);
                }
            }
            function heapSort(arr) {
                let n = arr.length;
                for (let i = Math.floor(n / 2) - 1; i >= 0; i--) {
                    heapify(arr, n, i);
                }
                for (let i = n - 1; i > 0; i--) {
                    [arr[i], arr[0]] = [arr[0], arr[i]];
                    heapify(arr, i, 0);
                }
            }
            let arr = [64, 34, 25, 12, 22, 11, 90];
            console.log("Before Heap Sort:");
            console.log(arr);
            heapSort(arr);
            console.log("After Heap Sort:");
            console.log(arr);
