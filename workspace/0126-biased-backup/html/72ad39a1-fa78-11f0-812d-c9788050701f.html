Type-Safe Function
TypeScript
function greet(name: string, age: number): string {
    return `Hello ${name}, you are ${age} years old!`;
}

// Valid usage
const message = greet("Alice", 28);

// Invalid usage (would show compile error)
// const error = greet(42, "Bob");

Custom Types
Rust
// Defining custom types
struct User {
    id: u32,
    username: String,
    is_premium: bool,
}

// Using the type
let user = User {
    id: 1,
    username: String::from("rust_lover"),
    is_premium: true,
};

// Pattern matching with types
match user.is_premium {
    true => println!("Premium user!"),
    false => println!("Regular user"),
}
