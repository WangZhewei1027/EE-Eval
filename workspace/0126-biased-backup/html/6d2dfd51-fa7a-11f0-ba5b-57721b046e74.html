function bucketSort(array, bucketCount, min, max) {
    // 1. Create empty buckets
    let buckets = new Array(bucketCount);
    for (let i = 0; i < bucketCount; i++) {
        buckets[i] = [];
    }
    
    // 2. Calculate range for each bucket
    let range = (max - min) / bucketCount;
    
    // 3. Scatter: Put array elements into buckets
    for (let num of array) {
        let bucketIndex = Math.floor((num - min) / range);
        if (bucketIndex >= bucketCount) bucketIndex = bucketCount - 1;
        buckets[bucketIndex].push(num);
    }
    
    // 4. Sort each bucket (using insertion sort)
    for (let bucket of buckets) {
        insertionSort(bucket);
    }
    
    // 5. Gather: Concatenate all buckets
    let sorted = [];
    for (let bucket of buckets) {
        sorted.push(...bucket);
    }
    
    return sorted;
}
